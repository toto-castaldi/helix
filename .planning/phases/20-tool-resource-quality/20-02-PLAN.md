**1a. Create the toolError helper function** in the Helper Functions section (after verifyClientOwnership, before MCP Protocol Implementation):

```typescript
// ============================================
// Error Helpers
// ============================================

type ErrorCategory = 'not_found' | 'access_denied' | 'validation_error' | 'database_error' | 'unknown_tool' | 'template_in_use'

function toolError(
  category: ErrorCategory,
  message: string
): { content: Array<{ type: string; text: string }>; isError: true } {
  return {
    content: [{ type: "text", text: `[${category}] ${message}` }],
    isError: true,
  }
}
```
