handleSubmit logic:
```typescript
const handleSubmit = async (e: React.FormEvent) => {
  e.preventDefault()
  setError(null)
  setIsSubmitting(true)

  try {
    const { data, error: invokeError } = await supabase.functions.invoke('docora-update-token', {
      body: {
        repositoryId: repository.id,
        newToken: token.trim(),
      },
    })

    if (invokeError) {
      setError(invokeError.message || 'Errore durante l\'aggiornamento del token')
      return
    }

    // Check for application-level error in response
    if (data?.error) {
      setError(data.error)
      return
    }

    // Success -- close dialog
    // Realtime subscription will update the card automatically
    onClose()
  } catch (err) {
    setError('Errore di connessione. Riprova.')
  } finally {
    setIsSubmitting(false)
  }
}
```
